import 'dart:io';
import 'package:flutter/material.dart';
import 'package:image_picker/image_picker.dart';
import 'package:tflite/tflite.dart';

void main() {
  runApp(CropScanApp());
}

class CropScanApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'CropScanAI',
      theme: ThemeData(primarySwatch: Colors.green),
      home: CropScanHome(),
    );
  }
}

class CropScanHome extends StatefulWidget {
  @override
  _CropScanHomeState createState() => _CropScanHomeState();
}

class _CropScanHomeState extends State<CropScanHome> {
  File? _image;
  List? _output;
  final picker = ImagePicker();

  @override
  void initState() {
    super.initState();
    loadModel();
  }

  Future loadModel() async {
    await Tflite.loadModel(
      model: "assets/model.tflite",
      labels: "assets/labels.txt",
    );
  }

  Future pickImage() async {
    final pickedFile = await picker.pickImage(source: ImageSource.gallery);
    if (pickedFile == null) return;

    setState(() => _image = File(pickedFile.path));
    runModel(_image!);
  }

  Future runModel(File image) async {
    final output = await Tflite.runModelOnImage(
      path: image.path,
      numResults: 2,
      threshold: 0.5,
      imageMean: 127.5,
      imageStd: 127.5,
    );
    setState(() => _output = output);
  }

  @override
  void dispose() {
    Tflite.close();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('CropScanAI')),
      body: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Column(
          children: [
            _image == null
                ? Text('Upload a crop image')
                : Image.file(_image!, height: 250),
            SizedBox(height: 20),
            _output != null
                ? Text(
                    "Detected: ${_output![0]["label"]}\nConfidence: ${(_output![0]["confidence"] * 100).toStringAsFixed(2)}%",
                    style: TextStyle(fontSize: 18),
                  )
                : Container(),
            Spacer(),
            ElevatedButton.icon(
              onPressed: pickImage,
              icon: Icon(Icons.image),
              label: Text('Upload Image'),
            ),
          ],
        ),
      ),
    );
  }
}
